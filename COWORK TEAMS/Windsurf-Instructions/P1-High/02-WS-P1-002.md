## REQUIRED CHANGE

**Before:**
```typescript
async function getDashboardStats(userId: string) {
  const stats = await db.query(
    `SELECT * FROM dispatch_stats WHERE owner_id = $1`, [userId]
  );
  return stats;
}
```

**After:**
```typescript
async function getDashboardStats(userId: string, req: Request) {
  await requireAccess(req, 'catalyst', targetCompanyId);
  const user = await getUser(userId);
  const stats = await db.query(
    `SELECT d.* FROM dispatch_stats d
     JOIN loads l ON d.load_id = l.id
     WHERE l.company_id = $1 AND d.owner_id = $2`,
    [user.companyId, userId]
  );
  return stats;
}
```
