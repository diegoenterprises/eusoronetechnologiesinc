## REQUIRED CHANGE

**In accessorial.ts approveAccessorialClaim():**
```typescript
async function approveAccessorialClaim(claimId: number, approverId: string) {
  const claim = await db.query(
    `SELECT * FROM accessorial_claims WHERE id = $1`, [claimId]
  );

  // 1. Update claim status
  await db.query(
    `UPDATE accessorial_claims SET status = 'approved', approved_by = $1,
     approved_at = NOW() WHERE id = $2`,
    [approverId, claimId]
  );

  // 2. Calculate fee and add to shipper invoice
  const platformFeePercent = 0.035;
  const grossAmount = claim.amount;
  const platformFee = Math.round(grossAmount * platformFeePercent * 100) / 100;
  const netAmount = grossAmount - platformFee;

  await db.query(
    `INSERT INTO invoice_line_items
     (invoice_id, description, amount, claim_id)
     VALUES ($1, $2, $3, $4)`,
    [claim.shipperId, `Accessorial: ${claim.type}`, grossAmount, claimId]
  );

  // 3. Record platform fee
  await db.query(
    `INSERT INTO platform_revenue (claim_id, fee_amount)
     VALUES ($1, $2)`,
    [claimId, platformFee]
  );

  // 4. Credit carrier wallet
  await db.query(
    `INSERT INTO wallet_transactions
     (user_id, amount, type, claim_id)
     VALUES ($1, $2, 'accessorial_credit', $3)`,
    [claim.carrierId, netAmount, claimId]
  );

  return { claim, platformFee, carrierCredit: netAmount };
}
```
