```typescript
import { messaging } from '../config/firebase';

interface PushNotification {
  title: string;
  body: string;
  data?: Record<string, string>;
}

async function sendPush(
  userId: string,
  notification: PushNotification
): Promise<void> {
  try {
    // Get user's push tokens
    const tokens = await db.query(
      `SELECT push_token FROM push_tokens WHERE user_id = $1 AND is_active = true`,
      [userId]
    );

    if (tokens.length === 0) {
      console.log(`[PUSH] No active tokens for user ${userId}`);
      return;
    }

    // Send to all active tokens
    const responses = await Promise.allSettled(
      tokens.map((token) =>
        messaging.send({
          token: token.pushToken,
          notification: {
            title: notification.title,
            body: notification.body,
          },
          data: notification.data || {},
          android: { priority: 'high' },
          apns: { headers: { 'apns-priority': '10' } },
        })
      )
    );

    // Log failures
    responses.forEach((result, idx) => {
      if (result.status === 'rejected') {
        console.warn(
          `[PUSH] Failed to send to token ${tokens[idx].pushToken}:`,
          result.reason
        );
        // Mark token as inactive if it fails
        db.query(
          `UPDATE push_tokens SET is_active = false WHERE push_token = $1`,
          [tokens[idx].pushToken]
        );
      }
    });

    console.log(`[PUSH] Sent to ${tokens.length} token(s)`);
  } catch (error) {
    console.error('[PUSH] Error sending notification:', error);
    throw error;
  }
}

// Example: Send load assignment notification
async function notifyLoadAssignment(driverId: string, loadId: number) {
  await sendPush(driverId, {
    title: 'New Load Assignment',
    body: `You've been assigned to load #${loadId}`,
    data: { loadId: loadId.toString(), action: 'load_detail' },
  });
}

export { sendPush, notifyLoadAssignment };
```
